                    data-containerSelector="body"
                    data-direction="top"
                    data-transition="ease"
                    data-duration="300"
                    data-clickSelector=".toggle-panel"
                >
                     <ul>
                        <li><a href="#">Home</a></li>
                        <li><a href="#">About</a></li>
                        <li><a href="#">Services</a></li>

                    </ul>
                </div>
